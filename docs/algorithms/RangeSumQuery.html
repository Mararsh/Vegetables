class NumArray {
  
    
    private int[] n, sum;

    public NumArray(int[] nums) {
        n = nums;
        if ( n == null || n.length == 0 ) return;
        sum = new int[n.length];
        sum[0] = n[0];
        for (int i=1; i < n.length; i++) {
          sum[i] = sum[i-1] + n[i];
        }
    }
    
    // Total runtime 2094 ms. Your submission beats 92.16% Submissions!
    public int sumRange(int i, int j) {
        if (n == null || i < 0 || j > n.length - 1) return 0;
        if ( i == 0 ) return sum[j];
        return sum[j] - sum[i-1];
    }
