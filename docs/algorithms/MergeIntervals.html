156. Merge Intervals

Given a collection of intervals, merge all overlapping intervals.
Example

Given intervals => merged intervals:

[                     [
  (1, 3),               (1, 6),
  (2, 6),      =>       (8, 10),
  (8, 10),              (15, 18)
  (15, 18)            ]
]

Challenge

O(n log n) time and O(1) extra space.

Idea to solve the problem

Sorting and set in-place

My codes




/**
 * Definition of Interval:
 * public classs Interval {
 *     int start, end;
 *     Interval(int start, int end) {
 *         this.start = start;
 *         this.end = end;
 *     }
 * }
 */

public class Solution {
    /**
     * @param intervals: interval list.
     * @return: A new interval list.
     */
    // Total runtime 2204  ms. Your submission beats 97.00% Submissions!
    public List merge(List intervals) {
        if ( intervals == null || intervals.size() < 2) return intervals;
        Collections.sort(intervals, new Comparator(){
            @Override
            public int compare(Interval a, Interval b) {
                return a.start - b.start;
            }
        });
        int index = 0;
        for (int i =1; i< intervals.size(); i++) {
            if ( intervals.get(i).start <= intervals.get(index).end  ) {
                if ( intervals.get(index).end <  intervals.get(i).end) {
                    intervals.get(index).end =  intervals.get(i).end;
                }
            } else {
                intervals.set(++index, intervals.get(i));
            }
        }
        return intervals.subList(0, index+1);
    }
}
