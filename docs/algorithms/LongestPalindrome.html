627. Longest Palindrome

Given a string which consists of lowercase or uppercase letters, 
find the length of the longest palindromes that can be built with those letters.

This is case sensitive, for example "Aa" is not considered a palindrome here.
Example

Given s = "abccccdd" return 7

One longest palindrome that can be built is "dccaccd", whose length is 7.

Idea to solve the problem

Using int[] as HashTable to count the numbers of the characters.

public class Solution {
    /**
     * @param s: a string which consists of lowercase or uppercase letters
     * @return: the length of the longest palindromes that can be built
     */
          // Total runtime 2676 ms. Your submission beats 99.80% Submissions!
      public int longestPalindrome(String s) {
        if (s == null) {
            return 0;
        }
        int[] m = new int[256];
        for (int i = 0; i < s.length(); i++) {
          m[s.charAt(i)]++;
        }
        boolean hasOdd = false;
        int count = 0;
        for (int i=0; i < m.length; i++) {
          if ( m[i] == 0 ) continue;
          if ( m[i] % 2 == 0 ) 
            count += m[i];
          else {
            hasOdd = true;
            count += m[i] - 1;
          }
        }
        if (hasOdd) {
            return count + 1;
        }
        return count;
    }

    public int longestPalindrome2(String s) {
        if (s == null) {
            return 0;
        }
        int result = 0;
        HashSet data = new HashSet<>();
        for (int i = 0; i < s.length(); i++) {
            char v = s.charAt(i);
            if (data.contains(v)) {
                result += 2;
                data.remove(v);
            } else {
                data.add(v);
            }
        }
        // if some chars left, then odd
        if (data.size() > 0) {
            return result + 1;
        }
        return result;
    }
    
    // Total runtime 2972 ms. Your submission beats 74.20% Submissions!
    public int longestPalindrome3(String s) {
        if (s == null) {
            return 0;
        }
        HashMap count = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (count.containsKey(c)) {
                count.put(c, count.get(c) + 1);
            } else {
                count.put(c, 1);
            }
        }
        int result = 0;
        boolean hasOdd = false;
        for (int value : count.values()) {
            if (value % 2 == 0) {
                result += value;
            } else {
                result += value - 1;
                hasOdd = true;
            }
        }
        if (hasOdd) {
            return result + 1;
        }
        return result;
    }
