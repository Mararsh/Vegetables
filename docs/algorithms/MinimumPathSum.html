110. Minimum Path Sum

Description of the Problem

Given a m x n grid filled with non-negative numbers, find a path from top left to bottom right 
which minimizes the sum of all numbers along its path.

Idea to solve the problem

Dynamic Programming

public class Solution {
    /**
     * @param grid: a list of lists of integers
     * @return: An integer, minimizes the sum of all numbers along its path
     */
    // Total runtime 2366 ms. Your submission beats 65.80% Submissions!
    public int minPathSum(int[][] grid) {
        if (grid == null || grid.length == 0 || grid[0].length == 0) return -1;
        int rows =  grid.length;
        int cols = grid[0].length;
        int[][] min = new int[rows][cols];
        min[0][0] = grid[0][0];
        for (int j=1; j< cols; j++) min[0][j] = grid[0][j] + min[0][j-1];
        for (int i=1; i< rows; i++) min[i][0] = grid[i][0] + min[i-1][0]; 
        for (int i=1; i< rows; i++) {
          for (int j=1; j<cols; j++) {
            min[i][j] = grid[i][j] + Math.min(min[i-1][j], min[i][j-1]);
          }
        }
        return min[rows-1][cols-1];
    }
}
