423. Valid Parentheses

Given a string containing just the characters '(', ')', '{', '}', '[' and ']', 
determine if the input string is valid.
Example

The brackets must close in the correct order, "()" and "()[]{}" are all valid but "(]" and "([)]" are not.

public class Solution {
    /**
     * @param s: A string
     * @return: whether the string is a valid parentheses
     */
    // Total runtime 3438 ms. Your submission beats 97.60% Submissions!
    public boolean isValidParentheses(String s) {
        if (s == null) {
            return false;
        }
        List left = Arrays.asList('(', '[', '{');
        List right = Arrays.asList(')', ']', '}');
        List stack = new ArrayList();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (left.contains(c)) {
                stack.add(c);
            } else if (right.contains(c)) {
                if ( stack.isEmpty() ) return false;
                char p = stack.get(stack.size() - 1);
                if ( c == ')' && p != '(') return false;
                if ( c == '}' && p != '{') return false;
                if ( c == ']' && p != '[') return false;
                stack.remove(stack.size() - 1);                
            }
        }
        return stack.isEmpty() ; 
    }

            } else if (right.contains(c)) {
                if ( stack.isEmpty() ) return false;
                String p = stack.get(stack.size() - 1);
                if ( c.equals(")") && !p.equals("(")) return false;
                if ( c.equals("]") && !p.equals("[")) return false;
                if ( c.equals("}") && !p.equals("{")) return false;
                stack.remove(stack.size() - 1);                
            }
        }
        return stack.isEmpty() ; 
    }
