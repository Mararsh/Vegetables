50. Product of Array Exclude Itself

Given an integers array A.

Define B[i] = A[0] * ... * A[i-1] * A[i+1] * ... * A[n-1], calculate B WITHOUT divide operation.
Example

For A = [1, 2, 3], return [6, 3, 2].

Idea to solve the problem

Dynamic Programming

My codes




public class Solution {
    /*
     * @param nums: Given an integers array A
     * @return: A long long array B and B[i]= A[0] * ... * A[i-1] * A[i+1] * ... * A[n-1]
     */
    // Total runtime 1822 ms. Your submission beats 99.60% Submissions!
    public List productExcludeItself(List nums) {
        int len = nums.size();
        long[] left = new long[len];
        long[] right = new long[len];
        left[0] = 1;
        right[len - 1] = 1;
        for (int i=1; i < len; i++) {
            left[i] = left[i-1] * nums.get(i-1);
            int j = len - 1 - i;
            right[j] = right[j+1] * nums.get(j+1);
        }
        List result = new ArrayList<>();
        for (int i=0; i < nums.size(); i++) {
            result.add( left[i] * right[i] );
        }
        return result;
    }
}
