    // Total runtime 2231 ms。Your submission beats 54.80% Submissions!
    public void insertSort(int[] A) {
        if ( A == null || A.length < 2 ) return;
        for (int i=1; i < A.length; i++) {
            int key = A[i], j = i-1;
            while (j >= 0 && A[j] > key ) {
                    A[j+1] = A[j--];
            }
            A[j+1] = key;
        }
    }
    
    // Total runtime 2397 ms. Your submission beats 52.60% Submissions!
    public void selection_sort(int[] A) {
        if ( A == null || A.length < 2) return;
        for (int i=0; i< A.length; i++) {
            for (int j=i+1; j <  A.length; j++) {
                if ( A[j] < A[i] ) {
                    int tmp = A[j];
                    A[j] = A[i];
                    A[i] = tmp;
                }
            }
        }
        
    }
    
    // Total runtime 2206 ms.。Your submission beats 57.80% Submissions!
    public void bubble_sort(int[] A) {
        if (A==null || A.length < 2) return;
        for (int i=A.length-1; i >= 0; i--) {
            for (int j=1; j<= i; j++) {
                if (A[j] < A[j-1]) {
                    int tmp = A[j-1];
                    A[j-1] = A[j];
                    A[j] = tmp;
                }
            }
        }
        
    }
