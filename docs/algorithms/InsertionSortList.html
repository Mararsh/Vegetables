173. Insertion Sort List

Sort a linked list using insertion sort.
Example

Given 1->3->2->0->null, return 0->1->2->3->null.

Idea to solve the problem

Insertion Sorting/Linked list operation

/**
 * Definition for ListNode
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */

public class Solution {
    /**
     * @param head: The first node of linked list.
     * @return: The head of linked list.
     */
    // Total runtime 3168 ms. Your submission beats 99.40% Submissions!
    public ListNode insertionSortList(ListNode head) {
        ListNode current = head.next, prev = head;
        while (current != null ) {
            ListNode next = current.next;
            if ( prev.val > current.val ) {
                prev.next = null;
                ListNode p = head;
                if ( current.val < head.val ) {
                    current.next = head;
                    head = current;
                } else {
                    while ( p.next != null && p.next.val <= current.val) {
                        p = p.next;
                    }
                    current.next = p.next;
                    p.next = current;
                }
                prev.next = next;
                current = next;
            } else {
                prev= current;
                current = next;
            }
            
        }
        return head;
    }
}
