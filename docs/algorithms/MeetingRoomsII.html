919. Meeting Rooms II

Given an array of meeting time intervals consisting of start and end times [[s1,e1],[s2,e2],...] (si < ei), 
find the minimum number of conference rooms required.
Example

Given intervals = [(0,30),(5,10),(15,20)], return 2.

/**
 * Definition of Interval:
 * public classs Interval {
 *     int start, end;
 *     Interval(int start, int end) {
 *         this.start = start;
 *         this.end = end;
 *     }

public class Solution {
    /**
     * @param intervals: an array of meeting time intervals
     * @return: the minimum number of conference rooms required
     */
    // Total runtime 1826  ms. Your submission beats 91.93% Submissions!
    public int minMeetingRooms(List intervals) {
        if (intervals == null) {
            return 0;
        }
        int len = intervals.size();
        int[] starts = new int[len];
        int[] ends = new int[len];
        for (int i = 0; i < len; i++) {
            starts[i] = intervals.get(i).start;
            ends[i] = intervals.get(i).end;
        }
        Arrays.sort(starts);
        Arrays.sort(ends);
        int sindex = 0, eindex = 0, max = 0, count = 0;
        while (sindex < len && eindex < len) {
            if (starts[sindex] < ends[eindex]) {
                count++;
                sindex++;
            } else if (starts[sindex] > ends[eindex]) {
                count--;
                eindex++;
            } else {
                sindex++;
                eindex++;
            }
            if (max < count) {
                max = count;
            }
        }
        return max;
    }
