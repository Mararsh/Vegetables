5. Kth Largest Element

Find K-th largest element in an array.
Example

In array [9,3,2,4,8], the 3rd largest element is 4.

In array [1,2,3,4,5], the 1st largest element is 5, 2nd largest element is 4, 3rd largest element is 3 and etc.
Challenge

O(n) time, O(1) extra memory.

Idea to solve the problem

Partition of quick sorting

My codes




class Solution {
    /*
     * @param k : description of k
     * @param nums : array of nums
     * @return: description of return
     */
     
    // Total runtime 910 ms. Your submission beats 68.00% Submissions! 
    public int kthLargestElement(int k, int[] nums) {
        if ( nums == null || nums.length ==0 || k > nums.length ) return -1;
        found = false;
        target = nums.length - k;
        quickSort(nums, 0, nums.length -1, k);
        return nums[target];
    }
    
    private boolean found = false;
    private int target = -1;
    
    private void quickSort(int[] nums, int left, int right, int k) {
        if (found || left >= right ) return;
        int pos = partition(nums, left, right);
        if ( pos == target) {
            found= true;
            return;
        }
        quickSort(nums, left, pos -1, k);
        quickSort(nums, pos + 1, right, k);
    }
    
    private int partition(int[] nums, int left, int right) {
        int key =  nums[left];
        int i=left, j=right;
        while (i < j ) {
            while ( i < j && nums[j] >= key) j--;
            if (i < j) nums[i++] = nums[j];
            while ( i < j && nums[i] <= key ) i++;
            if (i < j ) nums[j--] = nums[i];
        }
        nums[i] = key;
        return i;
    }
    
};
