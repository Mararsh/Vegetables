920. Meeting Rooms

Given an array of meeting time intervals consisting of start and end times [[s1,e1],[s2,e2],...] (si < ei), 
determine if a person could attend all meetings.

Example

Given intervals = [[0,30],[5,10],[15,20]], return false.

Idea to solve the problem

Customized sorting

My codes




/**
 * Definition of Interval:
 * public classs Interval {
 *     int start, end;
 *     Interval(int start, int end) {
 *         this.start = start;
 *         this.end = end;
 *     }
 * }
 */

public class Solution {
    /**
     * @param intervals: an array of meeting time intervals
     * @return: if a person could attend all meetings
     */
    // Total runtime 1973  ms. Your submission beats 86.52% Submissions!
    public boolean canAttendMeetings(List intervals) {
        if ( intervals == null || intervals.size() < 2 ) return true;
        Collections.sort(intervals, new Comparator(){
            @Override
            public int compare(Interval a, Interval b) {
                if ( a.start > b.start ) 
                    return 1;
                else if ( a.start < b.start )
                    return -1;
                else
                    return 0;
            }
        });

        for (int i=1; i < intervals.size(); i++) {
            if ( intervals.get(i).start < intervals.get(i-1).end ) return false;
        }
        return true;
    }
}
